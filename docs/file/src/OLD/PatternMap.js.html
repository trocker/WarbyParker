    import Pattern from './Pattern'

class PatternMap extends Map {

    add (pattern) {

        // See if the key is already in this map
        const key = this.get(pattern.toString())

        // If the key is found add the pattern to that key
        if (key) key.add(pattern)

        // If the key is not found add the pattern
        else {

            // Add the pattern to the super map
            super.add(pattern.toString(), pattern)

            // Make a copy of the patterns tokens
            const tokens = pattern.tokens.slice()

            // Remove the first token
            tokens.shift()

            // Create a new pattern
            const subPattern = new Pattern(pattern)

            // Set the sub patterns tokens
            subPattern.tokens = tokens

            // Add the sub pattern to this map
            this.add(subPattern)
        }
    }
